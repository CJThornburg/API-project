Table Users {
  id int [pk, increment]
  firstName varchar [not null]
  lastName varchar [not null]
  username varchar [not null, unique]
  hashedPassword varchar [not null]
  email varchar [not null, unique]
  createdAt timestamp
  updatedAt timestamp
}

Table Venues {
  id int [pk, increment]
  groupId int [ref: > Groups.id]
  address varchar
  city varchar
  state varchar
  lat decimal
  lng decimal
  createdAt timestamp
  updatedAt timestamp
}

Table Events {
  id int [pk, increment]
  venueId int [ref: > Venues.id]
  groupId int [ref: > Groups.id]
  name varchar [not null]
  description text
  type enum [not null]
  capacity int
  price int
  startDate date [not null]
  endDate date [not null]
  createdAt timestamp
  updatedAt timestamp
}

Table Groups {
  id int [pk, increment]
  organizerId int [ref: > Users.id]
  name varchar [not null]
  about text [not null]
  type enum [not null]
  private boolean [not null]
  city varchar
  state varchar
  createdAt timestamp
  updatedAt timestamp
}

Table GroupImages {
  id int [pk, increment]
  groupId int [ref: > Groups.id]
  url varchar
  preview boolean [not null, default: false]
  createdAt timestamp
  updatedAt timestamp
}

Table EventImages {
  id int [pk, increment]
  eventId int [ref: > Events.id]
  url varchar
  preview boolean [not null, default: false]
  createdAt timestamp
  updatedAt timestamp
}

Table Memberships {
  id int [pk, increment]
  userId int [ref: > Users.id]
  groupId int [ref: > Groups.id]
  status enum [not null]
  createdAt timestamp
  updatedAt timestamp
}

Table Attendances {
  id int [pk, increment]
  eventId int [ref: > Events.id]
  userId int [ref: > Users.id]
  status enum [not null]
  createdAt timestamp
  updatedAt timestamp
}
